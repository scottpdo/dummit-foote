\section*{1. (2/16/23)}

\begin{itemize}
    \item $\sigma: (1, 3, 5)(2, 4)$
    \item $\tau: (1, 5)(2, 3)$
    \item $\sigma^2: (1, 5, 3)$
    \item $\sigma \tau: (2, 5, 3, 4)$
    \item $\tau \sigma: (1, 2, 4, 3)$
    \item $\tau^2 \sigma: (1, 3, 5)(2, 4)$ (because $\tau^2 = 1$, so $\tau^2 \sigma = \sigma$)
\end{itemize}
