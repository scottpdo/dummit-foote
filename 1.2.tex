\section*{1. (1/23/23)}

Compute the order of each of the elements in the following groups:

\begin{enumerate}[label=(\alph*)]
    \item $D_6$
        \begin{itemize}
            \item $r, r^2$: 3
            \item $s, sr, sr^2$: 2
        \end{itemize}
    
    \item $D_8$
        \begin{itemize}
            \item $r$: 4
            \item $r^2$: 2
            \item $r^3$: 4
            \item $s, sr, sr^2, sr^3$: 2
        \end{itemize}

    \item $D_{10}$
        \begin{itemize}
            \item $r, r^2, r^3, r^4$: 5
            \item $s, sr, sr^2, sr^3, sr^4$: 2
        \end{itemize}
    
\end{enumerate}
